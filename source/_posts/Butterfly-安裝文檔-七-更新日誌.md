   ```diff
   -preloader: false
   +preloader:
   +  enable: false
   +  # source
   +  # 1. fullpage-loading
   +  # 2. pace (progress bar)
   +  source: 1
   +  # pace theme (see https://codebyzach.github.io/pace/)
   +  pace_css_url:

   -anchor: false
   +anchor:
   +  button:
   +    enable: false
   +    always_show: false
   +    icon: # the unicode value of Font Awesome icon, such as '\3423'
   +  auto_update: false # when you scroll in post, the URL will update according to header id.
   ```
